72  public DataSet next(int num) {
73  if (exampleStartOffsets.size() == 0) throw new NoSuchElementException();
74  int actualMiniBatchSize = Math.min(num, exampleStartOffsets.size());
75  INDArray input = Nd4j.create(new int[] {actualMiniBatchSize, VECTOR_SIZE, exampleLength}, 'f');

80  else
81  label = Nd4j.create(new int[] {actualMiniBatchSize, predictLength, exampleLength}, 'f');
82 
83  for (int index = 0; index < actualMiniBatchSize; index++) {
84  int startIdx = exampleStartOffsets.removeFirst();

87  StockData nextData;
88  for (int i = startIdx; i < endIdx; i++) {

91  input.putScalar(new int[] {index, 1, c}, (curData.getClose() - minArray[1]) / (maxArray[1] - minArray[1]));
92  input.putScalar(new int[] {index, 2, c}, (curData.getLow() - minArray[2]) / (maxArray[2] - minArray[2]));
93  input.putScalar(new int[] {index, 3, c}, (curData.getHigh() - minArray[3]) / (maxArray[3] - minArray[3]));
94  input.putScalar(new int[] {index, 4, c}, (curData.getVolume() - minArray[4]) / (maxArray[4] - minArray[4]));
95  nextData = train.get(i + 1);
96  if (category.equals(PriceCategory.ALL)) {
97  label.putScalar(new int[] {index, 0, c}, (nextData.getOpen() - minArray[1]) / (maxArray[1] - minArray[1]));
98  label.putScalar(new int[] {index, 1, c}, (nextData.getClose() - minArray[1]) / (maxArray[1] - minArray[1]));
99  label.putScalar(new int[] {index, 2, c}, (nextData.getLow() - minArray[2]) / (maxArray[2] - minArray[2]));
100  label.putScalar(new int[] {index, 3, c}, (nextData.getHigh() - minArray[3]) / (maxArray[3] - minArray[3]));
101  label.putScalar(new int[] {index, 4, c}, (nextData.getVolume() - minArray[4]) / (maxArray[4] - minArray[4]));

120  default: throw new NoSuchElementException();

158  private List<Pair<INDArray, INDArray>> generateTestDataSet (List<StockData> stockDataList) {
159  int window = exampleLength + predictLength;
160  List<Pair<INDArray, INDArray>> test = new ArrayList<>();

163  for (int j = i; j < i + exampleLength; j++) {
164  StockData stock = stockDataList.get(j);
165  input.putScalar(new int[] {j - i, 0}, (stock.getOpen() - minArray[0]) / (maxArray[0] - minArray[0]));
166  input.putScalar(new int[] {j - i, 1}, (stock.getClose() - minArray[1]) / (maxArray[1] - minArray[1]));
167  input.putScalar(new int[] {j - i, 2}, (stock.getLow() - minArray[2]) / (maxArray[2] - minArray[2]));
168  input.putScalar(new int[] {j - i, 3}, (stock.getHigh() - minArray[3]) / (maxArray[3] - minArray[3]));
169  input.putScalar(new int[] {j - i, 4}, (stock.getVolume() - minArray[4]) / (maxArray[4] - minArray[4]));
170  }
171  StockData stock = stockDataList.get(i + exampleLength);
172  INDArray label;
173  if (category.equals(PriceCategory.ALL)) {
174  label = Nd4j.create(new int[]{VECTOR_SIZE}, 'f'); // ordering is set as 'f', faster construct
175  label.putScalar(new int[] {0}, stock.getOpen());

179  label.putScalar(new int[] {4}, stock.getVolume());

184  case CLOSE: label.putScalar(new int[] {0}, stock.getClose()); break;
185  case LOW: label.putScalar(new int[] {0}, stock.getLow()); break;

197  private List<StockData> readStockDataFromFile (String filename, String symbol) {
198  List<StockData> stockDataList = new ArrayList<>();
199  try {
200  for (int i = 0; i < maxArray.length; i++) { // initialize max and min arrays
201  maxArray[i] = Double.MIN_VALUE;
202  minArray[i] = Double.MAX_VALUE;
203  }
204  List<String[]> list = new CSVReader(new FileReader(filename)).readAll(); // load all elements in a list
205  for (String[] arr : list) {
206  if (!arr[1].equals(symbol)) continue;
207  double[] nums = new double[VECTOR_SIZE];
208  for (int i = 0; i < arr.length - 2; i++) {
209  nums[i] = Double.valueOf(arr[i + 2]);
210  if (nums[i] > maxArray[i]) maxArray[i] = nums[i];
211  if (nums[i] < minArray[i]) minArray[i] = nums[i];
212  }
213  stockDataList.add(new StockData(arr[0], arr[1], nums[0], nums[1], nums[2], nums[3], nums[4]));
214  }
215  } catch (IOException e) {
216  e.printStackTrace();
217  }
218  return stockDataList;
219  }
